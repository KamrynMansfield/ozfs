National Zoning Atlas to OZFS Code

The data we used was already in a computer friendly format following the National Zoning Atlas guidelines. This page explains in detail the r functions made to convert the NZA data to OZFS format.

Overview

Functions

The list below shows the functions we created. The indentations show functions used in other functions.

  • atlas_to_ozfs
    • organize_features
      • separate_uses
      • add_uses_permitted_column
      • update_uses_permitted
      • make_constraints_section
        • organize_rules
  • write_list_as_json
  • add_geometry_to_ozfs

Inputs

Necessary inputs:

  • atlas_df: a data frame created from an excel file that has isolated the NZA data for just one city. This takes some manual formatting
  • col_descriptions: a data frame created from an excel file that lists all the column names and some descriptions of those columns. (like the units and whether it is min or max value )
  • use_type_indicators: a unique list I made fad
  • boundary_file_path: a path to the geojson file containing the district boundary geometry of the city

Optional Inputs

  • created_by: The name of the person who first converted the data into OZFS
  • creation_date: The date the data was first converted the into OZFS
  • updated_by: The name of the person who updated the OZFS structure

atlas_to_ozfs

This is the main function that is used to convert the data from the excel file (formatted according to NZA standards) to a geojson file (formatted according to OZFS standards).
Code 
atlas_to_ozfs <- function(atlas_df, col_descriptions, use_type_indicators,
                          created_by = "Kamryn Mansfield", creation_date = "2024-09-30",
                          updated_by = "Kamryn Mansfield"){
  
  # start the empty list
  ozfs_format <- list(type = "FeatureCollection",
                      metadata = list(muni_name = atlas_df[[1,"muni_name"]],
                                      county_name = atlas_df[[1, "county"]]),
                      features = list())
  
  # loop through each row of atlas_df
  for (i in 1:nrow(atlas_df)){
    atlas_row_df <- atlas_df[i,]
    ozfs_format$features[[i]] <- organize_features(atlas_row_df, col_descriptions, use_type_indicators,
                                                   created_by, creation_date,
                                                   updated_by)
    
  }
  
  ozfs_format
}

organize_features <- function(atlas_row_df, col_descriptions, use_type_indicators, created_by, creation_date, updated_by){
  #start with a bare list for the features data to fill
  features_list <- list(type = "Feature", 
                        properties = list(),
                        geometry = list())
  
  # make a df for each land use
  wanted_columns <- col_descriptions$col_name
  separated_uses <- separate_uses(atlas_row_df, use_type_indicators, wanted_columns)
  
  # fill out the dist_info section
  features_list$properties$dist_info[["dist_name"]] <- atlas_row_df[[1,"dist_name"]]
  features_list$properties$dist_info[["dist_abbr"]] <- atlas_row_df[[1,"dist_abbr"]]
  features_list$properties$dist_info[["dist_type"]] <- atlas_row_df[[1,"dist_type"]]
  features_list$properties$dist_info[["created_by"]] <- created_by
  features_list$properties$dist_info[["created_date"]] <- creation_date
  features_list$properties$dist_info[["updated_by"]] <- updated_by
  features_list$properties$dist_info[["updated_date"]] <- as.character(Sys.Date())
  
  # fill out uses permitted
  updated_atlas_row_df <- add_uses_permitted_column(atlas_row_df, use_type_indicators)
  features_list <- update_uses_permitted(features_list,updated_atlas_row_df)
  
  # prepare to loop through the other three sections
  lot_constraints <- col_descriptions |>
    filter(col_name %in% substr(names(atlas_row_df), 1, nchar(names(atlas_row_df)) - 2)) |>
    filter(heading == "lot_constraints")
  structure_constraints <- col_descriptions |>
    filter(col_name %in% substr(names(atlas_row_df), 1, nchar(names(atlas_row_df)) - 2)) |>
    filter(heading == "structure_constraints")
  other_constraints <- col_descriptions |>
    filter(col_name %in% substr(names(atlas_row_df), 1, nchar(names(atlas_row_df)) - 2)) |>
    filter(heading == "other_constraints")
  
  features_list$properties$lot_constraints <- make_constraints_section(lot_constraints, "lot_constraints", separated_uses)
  
  features_list$properties$structure_constraints <- make_constraints_section(structure_constraints, "structure_constraints", separated_uses)
  
  features_list$properties$other_constraints <- make_constraints_section(other_constraints, "other_constraints", separated_uses)
  
  
  features_list
  
  # add the uses permitted columns df
  muni_atlas_df <- add_uses_permitted_column(atlas_row_df, 
                                             use_type_indicators)
  # finish by adding the lists of uses into the district info
  features_list <- update_uses_permitted(features_list, 
                                     muni_atlas_df)
  
  features_list
}

separate_uses <- function(df, use_type_indicators, wanted_columns){
  list_of_use_tables <- list()
  for (i in 1:length(use_type_indicators)){
    cols_we_want <- grep(use_type_indicators[[i]], names(df), value = T)
    new_df <- df |>
      select(all_of(cols_we_want))
    
    names(new_df) <- str_replace_all(names(new_df), use_type_indicators[[i]] , "")
    
    # new_df <- new_df |>
    #   select(any_of(wanted_columns))
    
    list_of_use_tables[[names(use_type_indicators[i])]] <- new_df
  }
  
  list_of_use_tables
  
}

add_uses_permitted_column

# adding columns with lists of uses_permitted and uses_permitted_exception
add_uses_permitted_column <- function(df, use_type_indicators){
  
  df <- mutate(df, uses_permitted = vector("list", n()),
               uses_permitted_exception = vector("list", n()))
  uses_permitted_df <- df |>
    select(dist_abbr,all_of(grep("use_permitted",names(df), value = T)))
  
  for (j in 1:nrow(df)){
    uses_permitted_list <- list()
    uses_exception_list <- list()
    
    for (i in 1:length(use_type_indicators)){
      cols_we_want <- grep(use_type_indicators[[i]], names(uses_permitted_df), value = T)
      field_to_check <- uses_permitted_df[[j,cols_we_want]]
      if (!is.na(field_to_check) & field_to_check == "Allowed/Conditional"){
        uses_permitted_list <- append(uses_permitted_list, names(use_type_indicators[i]))
        uses_exception_list <- append(uses_exception_list, names(use_type_indicators[i]))
      } else if(!is.na(field_to_check) & field_to_check == "Public Hearing"){
        uses_exception_list <- append(uses_exception_list, names(use_type_indicators[i]))
      }
    }
    df$uses_permitted[[j]] <- uses_permitted_list
    df$uses_permitted_exception[[j]] <- uses_exception_list
  }
  df
}

# takes in the geojson formatted data and 
# the updated df that has the uses_permitted lists in it
update_uses_permitted <- function(geojson_formatted_data, updated_df){

    # name the variable that I will be using a bit to plug in and test things.
    uses_permitted <- updated_df[[1, "uses_permitted"]][[1]]
    uses_permit_exception <- updated_df[[1, "uses_permitted_exception"]][[1]]
    #replace the geojson_formatted_data with the value in the updated_df
    if (length(uses_permitted) > 0){
      geojson_formatted_data$properties$dist_info$uses_permitted$uses_value <- uses_permitted
    } else {
      geojson_formatted_data$properties$dist_info$uses_permitted$uses_value <- "none"
    }
    #find out if there is an exception and add that
    if (length(uses_permit_exception) > 0 & !identical(uses_permit_exception, uses_permitted)){
      geojson_formatted_data$properties$dist_info$uses_permitted$excep_allowed <- TRUE
      geojson_formatted_data$properties$dist_info$uses_permitted$excep_value <- uses_permit_exception
    } else {
      geojson_formatted_data$properties$dist_info$uses_permitted$excep_allowed <- FALSE
    }
  
  geojson_formatted_data
}

make_constraints_section <- function(constraints_df, constraints_section, separated_uses){

  # start the empty list
  constraint_list <- list() 
  # loop through the constraints section
  for (i in 1:nrow(constraints_df)){
    constraint_name <- constraints_df[[i,"col_name"]]
    
    # loop through each separated df and list the uses and their values
    use_name <- c()
    value <- c()
    for (j in 1:length(separated_uses)){
      
      # one of the separated land use dfs wiht only the columns pertaining to the constraint we are on
      separated_use_df <- separated_uses[[j]][,grep(constraint_name,names(separated_uses[[j]]))]
      # the name of the active df
      separated_use_name <- names(separated_uses[j])
      
      # create variables to hold the info
      use_name <- c(use_name, separated_use_name)
      # not every use has the column, so we need to add NA
      if (ncol(separated_use_df) == 1){
        value <- c(value, as.character(separated_use_df[[1, 1]]))
      } else if (ncol(separated_use_df) > 1){ 
        value <- c(value, as.character(separated_use_df[[1, 2]]))
      }else {
        value <- c(value, NA)
      }
      
    }
    
    
    # take uses and values to create 
    # a df with values and land uses that share that value
    values_uses_df <- data.frame(use_name = use_name,
                                 value = value) |>
      group_by(value) |>
      summarise(use_name = list(use_name)) |> 
      filter(!is.na(value))

    
    if (nrow(values_uses_df) > 0){
      
      
      # loop through the values_uses_df to add values to the geojson
      for (l in 1:nrow(values_uses_df)){
        
        # add length to the list so it can hold multiple things
        if (!is.null(constraint_list[[constraint_name]])) {
          length(constraint_list[[constraint_name]]) <- l
        }
      
        
        # add the units
        constraint_list[[constraint_name]][[l]]$unit <- constraints_df[[i,"units"]]
        # adding the use_name
        constraint_list[[constraint_name]][[l]]$use_name <- values_uses_df$use_name[[l]]
        
        # min or max value
        max_min_field <- constraints_df[[i,"min_or_max"]]
        if (!is.na(max_min_field) & max_min_field == "min"){
          values_uses_first_use <- values_uses_df$use_name[[l]][[1]]
          df_with_rules <- separated_uses[[values_uses_first_use]][,grep(constraint_name,names(separated_uses[[values_uses_first_use]]))]
          if (ncol(df_with_rules) > 1){
            constraint_list[[constraint_name]][[l]]$min_val <- organize_rules(df_with_rules)
          } else {
            constraint_list[[constraint_name]][[l]]$min_val$expression <- as.character(df_with_rules[[1,1]])
          }
          
        } else if (!is.na(max_min_field) & max_min_field == "max"){
          values_uses_first_use <- values_uses_df$use_name[[l]][[1]]
          df_with_rules <- separated_uses[[values_uses_first_use]][,grep(constraint_name,names(separated_uses[[values_uses_first_use]]))]
          if (ncol(df_with_rules) > 1){
            constraint_list[[constraint_name]][[l]]$max_val <- organize_rules(df_with_rules)
          } else {
            constraint_list[[constraint_name]][[l]]$max_val$expression <- as.character(df_with_rules[[1,1]])
          }
        }
        
      }
    }
    
  
    
  }
  constraint_list
}

organize_rules

#### THIS ORGANIZES THE LISTS UNDER CONSTRAINT VALUES ####
# df_with_rules = dataframe with all the columns 
#   for the rules of one of the variables selected.
# This dataframe must have ncol > 1
organize_rules <- function(df_with_rules){
  
  # find out how many rules there are
  counter <- 1
  df <- df_with_rules[1,grep(paste0("rule", counter), names(df_with_rules))]
  while (ncol(df) > 0){
    counter <- counter + 1
    df <- df_with_rules[1 ,grep(paste0("rule", counter), names(df_with_rules))]
  }
  
  # loop through each rule and make it a list
  all_rule_list <- list()
  for (i in 1:(counter - 1)){
    # create a list that we will keep adding to
    rule_list <- list()
    
    # New df with an isolated rule
    rule_df <- df_with_rules[ ,grep(paste0("rule", i), names(df_with_rules))]
    
    # if it has one of the fields, we will add it to rule_list
    
    # logical_operator
    if (sum(grep("logical_operator", names(rule_df))) > 0 ){
      # assign value to logical_operator
      rule_list$logical_operator <- rule_df[[1, grep("logical_operator",names(rule_df))]]
    }
    
    # conditions
    if (sum(grep("condition", names(rule_df))) > 0 ){
      # make a df to assign multiple values to the array of conditions
      condition_df <- rule_df[ , grep("condition",names(rule_df))]
      
      for (j in 1:ncol(condition_df)){
        rule_list$conditions[[j]] <- condition_df[[1,j]]
      }
    }
    
    # select
    if (sum(grep("select", names(rule_df))) > 0 ){
      # assign value to select
      rule_list$select <- rule_df[[1, grep("select",names(rule_df))]]
    }
    
    # expression(s)
    if (sum(grep("expression", names(rule_df))) > 0 ){
      # make a df to see if it is more than one expression and to extract data
      df_expression <- rule_df[ , grep(paste0("expression"), names(rule_df))]
      
      if (ncol(df_expression) > 1){ # if it is multiple expressions
        
        for (j in 1:ncol(df_expression)){
          rule_list$expressions[[j]] <- df_expression[[1,j]]
        }
        
      } else { # if it is just one expression
        # assign the value to the expression
        rule_list$expression <- rule_df[[1, grep("expression",names(rule_df))]]
      }
      
    }
    # add each rule list to the total rules list
    all_rule_list[[i]] <- rule_list
  }
  
  all_rule_list  
}

add_geometry_to_ozfs <- function(boundary_file_path, ozfs_list){
  boundaries <- rjson::fromJSON(file = boundary_file_path)
  for (i in 1:length(ozfs_list$features)){
    for (j in 1:length(boundaries$features)){
      if (ozfs_list$features[[i]]$properties$dist_info$dist_abbr == boundaries$features[[j]]$properties$`Abbreviated District Name`){
        ozfs_list$features[[i]]$geometry <- boundaries$features[[j]]$geometry
      }
    }
  }
  ozfs_list
}

make_ozfs <- function(list_nza_excels, list_geometries, new_folder_to_save_to){
  col_descriptions <- read_excel("files/atlas_to_ozfs/col_descriptions.xlsx")
  use_type_indicators <- list(`1_family` = "_1", 
                            `2_family` = "_2",
                            `3_family` = "_3",
                            `4_family` = "_4",
                            `ADU` = "_adu",
                            `Affordable_Housing` = "_ah",
                            `Planned_Residential_Development` = "_prd")
  errors <- c()
  error_messages <- c()
  for (i in 1:length(list_nza_excels)){
    nza_excel <- read_excel(list_nza_excels[[i]])
    
    file_name <- basename(list_nza_excels[[i]])
    file_name_no_ext <- sub(".xlsx","",file_name)
    
    ozfs_list_format <- tryCatch(
  {
    # Code that might throw an error
    atlas_to_ozfs(nza_excel, col_descriptions, use_type_indicators)
  },
  error = function(e) {
    # Code to run if an error occurs
    errors <- c(errors, list_nza_excels[[i]])
    error_messages <- c(error_messages, e$message)
    next
  }
)
    ozfs_list_with_geom <- add_geometry_to_ozfs(list_geometries[[i]], ozfs_list_format)
    
    new_file_directory <- paste0(new_folder_to_save_to,city_name, file_name_no_ext, ".geojson")
    write_list_as_json(ozfs_list_format, new_file_directory)
  }
}
